# 对着自机飞同时打自机狙的杂鱼，用作道中增援

PRESET_ENEMY_E
SET_HITPOINT 5
SET_MAXGETPOINTADD 5
SET_INVINCIBLE 10

CALC_DIRECTION 0 %ENEMY_X% %ENEMY_Y% %PLAYER_X% %PLAYER_Y%
RANDOM_FLOAT 1 -20 20
ADD 0 %VAR_1%
SET_ANGLE %VAR_0%
RANDOM_FLOAT 1 2 3
SET_SPEED %VAR_1%

WAIT 15

SFX ENEMY_SHOOT_A @ rPxvMUMw
SHOOT_AIMING TYPE_A 0 0 3.5 0
WAIT 240
JUMP_L %ENEMY_Y% 160 @rPxvMUMw

HALT
