# 往正下飞到1/3位置，打几发自机狙然后向两边撤退的杂鱼

PRESET_ENEMY_E
SET_HITPOINT 15
SET_INVINCIBLE 30

# 判断是在左边还有右边生成，使用%VAR_7%保存转向角度
JUMP_LE %ENEMY_X% 192 @VpJjcjZt
STORE 7 10
JUMP_GE %ENEMY_X% 192 @lRNYMrTp
STORE 7 -10 @ VpJjcjZt
NOOP @ lRNYMrTp

MOVE_RELATIVE 0 90 30 LINEAR
WAIT 30
MOVE_RELATIVE 0 30 30 EASE_OUT_CUBIC
WAIT 30

SFX ENEMY_SHOOT_A
SHOOT_AIMING TYPE_A 0 0 3 0
STORE 0 0
INC 0 @ wgAgwZRI
SET_ANGLE_RELATIVE %VAR_7%
WAIT 3
JUMP_L %VAR_0% 10 @wgAgwZRI

SET_SPEED 2

STORE 0 0
INC 0 @ W2h32SGA
SFX ENEMY_SHOOT_A
SHOOT_AIMING TYPE_A 0 0 3 0
WAIT 80
JUMP_L %VAR_0% 3 @W2h32SGA

HALT
