# 道中杂鱼增援，飞到左右1/3位置打两发开花弹然后飞走

PRESET_ENEMY_E
SET_HITPOINT 15
SET_INVINCIBLE 15

# 判断是在左边还有右边生成，使用%VAR_7%保存转向角度
JUMP_LE %ENEMY_X% 192 @VpJjcjZt
STORE 7 4
JUMP_GE %ENEMY_X% 192 @lRNYMrTp
STORE 7 -4 @ VpJjcjZt
NOOP @ lRNYMrTp

SET_SPEED 3
WAIT 45

SET_SPEED 0
STORE 0 0
INC 0 @ BSLPytdV
    STORE 1 %VAR_0%
    MOD 1 7
    JUMP_NE %VAR_1% 0 @VHJDLwrx
    SFX ENEMY_SHOOT_A
    SHOOT_AIMING_MULTIWAY TYPE_A 0 0 2.5 0 360 11
    SET_ANGLE_RELATIVE %VAR_7% @ VHJDLwrx
    WAIT 2
JUMP_L %VAR_0% 15 @BSLPytdV

SET_SPEED 4

HALT
